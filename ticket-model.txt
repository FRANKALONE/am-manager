
model Ticket {
  id                Int      @id @default(autoincrement())
  workPackageId     String
  issueKey          String   // JIRA ticket key (e.g., IMP-123)
  issueSummary      String
  issueType         String   // Type of ticket (Incidencia, Correctivo, Consulta, etc.)
  createdDate       DateTime // When the ticket was created in JIRA
  year              Int      // Year of creation (for easy filtering)
  month             Int      // Month of creation (for easy filtering)
  
  workPackage       WorkPackage @relation(fields: [workPackageId], references: [id], onDelete: Cascade)
  
  @@unique([workPackageId, issueKey]) // Prevent duplicates
  @@index([workPackageId, year, month])
}
